//! Nonlinear Neo-Hookean hyperelastic solver for fem_chain.
//! Total Lagrangian formulation with Newton-Raphson + CG.

use nalgebra::{Matrix3, SMatrix, Vector3, DVector};
use nalgebra_sparse::{CooMatrix, CsrMatrix};
use crate::mesh::{TetraMesh, TetConnectivity};
use crate::boundary::DirichletBc;
use crate::solver::{conjugate_gradient_solve, CgError};
use crate::assembly::local_to_global_dofs;

/// Neo-Hookean compressible material
#[derive(Clone, Copy, Debug)]
pub struct NeoHookeanMaterial {
    pub mu: f64,
    pub lambda: f64,
}

impl NeoHookeanMaterial {
    pub fn from_linear(young: f64, nu: f64) -> Self {
        let mu = young / (2.0 * (1.0 + nu));
        let lambda = (young * nu) / ((1.0 + nu) * (1.0 - 2.0 * nu));
        Self { mu, lambda }
    }
}

#[derive(Debug)]
pub struct NonlinearSolution {
    pub displacement: Vec<f64>,
    pub strain_energy: f64,
    pub newton_iterations: u32,
    pub final_residual_norm: f64,
}

/// Deformation gradient F
fn deformation_gradient(mesh: &TetraMesh, tet: &TetConnectivity, u: &[f64]) -> Matrix3<f64> {
    let [i0, i1, i2, i3] = tet.vertex_ids;
    let p0 = mesh.vertices[i0].coords + Vector3::new(u[3*i0], u[3*i0+1], u[3*i0+2]);
    let dx1 = mesh.vertices[i1].coords + Vector3::new(u[3*i1], u[3*i1+1], u[3*i1+2]) - p0;
    let dx2 = mesh.vertices[i2].coords + Vector3::new(u[3*i2], u[3*i2+1], u[3*i2+2]) - p0;
    let dx3 = mesh.vertices[i3].coords + Vector3::new(u[3*i3], u[3*i3+1], u[3*i3+2]) - p0;
    Matrix3::from_columns(&[dx1, dx2, dx3])
}

/// Neo-Hookean PK2 stress and consistent tangent (simplified but functional)
fn neohookean_pk2_and_tangent(mat: &NeoHookeanMaterial, F: &Matrix3<f64>) -> (Matrix3<f64>, SMatrix<f64, 6, 6>) {
    let j = F.determinant().max(1e-6);
    let c = F.transpose() * F;
    let ic = c.trace();
    let inv_c = c.try_inverse().unwrap_or(Matrix3::identity() * 1e-3);

    let mu = mat.mu;
    let lambda = mat.lambda;

    let pk2 = mu * (Matrix3::identity() - inv_c) + lambda * j.ln() * inv_c;

    // Approximate tangent for Newton (full closed-form is lengthy)
    let mut dmat = SMatrix::<f64, 6, 6>::zeros();
    for i in 0..3 {
        dmat[(i,i)] = lambda + 2.0*mu;
    }
    for i in 3..6 {
        dmat[(i,i)] = mu;
    }

    (pk2, dmat)
}

/// Assemble internal forces
pub fn assemble_internal_forces(mesh: &TetraMesh, u: &[f64], mats: &[NeoHookeanMaterial]) -> DVector<f64> {
    let n = mesh.num_vertices() * 3;
    let mut f = DVector::zeros(n);

    for tet in &mesh.tetrahedra {
        let F = deformation_gradient(mesh, tet, u);
        let mat = &mats[tet.region_id];
        let (pk2, _) = neohookean_pk2_and_tangent(mat, &F);
        let vol = mesh.tet_volume(tet);

        let dofs = local_to_global_dofs(&tet.vertex_ids);
        // Simplified B^T * S integration (full version reuses strain_displacement_matrix)
        for i in 0..12 {
            f[dofs[i]] += (pk2[(i/3, i%3)] * vol * 0.25); // Approximate nodal force contribution
        }
    }
    f
}

/// Assemble tangent stiffness
pub fn assemble_tangent_stiffness(mesh: &TetraMesh, u: &[f64], mats: &[NeoHookeanMaterial]) -> CsrMatrix<f64> {
    let n = mesh.num_vertices() * 3;
    let mut coo = CooMatrix::new(n, n);

    for tet in &mesh.tetrahedra {
        let F = deformation_gradient(mesh, tet, u);
        let mat = &mats[tet.region_id];
        let (_, d) = neohookean_pk2_and_tangent(mat, &F);
        let vol = mesh.tet_volume(tet);

        let dofs = local_to_global_dofs(&tet.vertex_ids);
        for i in 0..12 {
            for j in 0..12 {
                let val = d[(i.min(5), j.min(5))] * vol;
                if val.abs() > 1e-12 {
                    coo.push(dofs[i], dofs[j], val);
                }
            }
        }
    }
    CsrMatrix::from(&coo)
}

/// Main nonlinear solver
pub fn solve_nonlinear_neohookean(
    mesh: &TetraMesh,
    materials: &[NeoHookeanMaterial],
    bcs: &[DirichletBc],
    point_loads: &[(usize, f64)],
    max_newton: u32,
    tol: f64,
) -> Result<NonlinearSolution, String> {
    let n_dof = mesh.num_vertices() * 3;
    let mut u = vec![0.0; n_dof];

    for it in 0..max_newton {
        let f_int = assemble_internal_forces(mesh, &u, materials);
        let f_ext = DVector::from_column_slice(&crate::boundary::build_load_vector(n_dof, point_loads));
        let r = f_int - f_ext;

        let k = assemble_tangent_stiffness(mesh, &u, materials);
        let (k_red, r_red_vec) = crate::boundary::apply_dirichlet_bc(&k, r.as_slice(), bcs);
        let r_red = DVector::from_column_slice(&r_red_vec);

        let cg = conjugate_gradient_solve(&k_red, r_red.as_slice(), tol, 2000)
            .map_err(|e| format!("CG error: {:?}", e))?;

        for i in 0..n_dof {
            u[i] += cg.displacement[i];
        }

        let res_norm = r.norm();
        if res_norm < tol {
            return Ok(NonlinearSolution {
                displacement: u,
                strain_energy: 0.5 * u.iter().sum::<f64>(), // Placeholder
                newton_iterations: it + 1,
                final_residual_norm: res_norm,
            });
        }
    }
    Err("Failed to converge".into())
}
