//! Couples the agent population to a `ConcentrationField`: each tick,
//! agents move and release part of their decaying payload into the field
//! at their (post-motion) position, then the field diffuses. This is what
//! turns two independently-tested pieces (`Agent`/`Engine` and
//! `ConcentrationField`) into one simulation where agents actually affect
//! a shared environment instead of moving in isolation.

use crate::agent::{Agent, AgentParams};
use crate::concentration::ConcentrationField;
use crate::environment::Bounds;
use rand::Rng;

#[derive(Debug, Clone, Copy)]
pub struct CoupledTickStats {
    pub tick: u64,
    pub total_agent_payload: f64,
    pub total_field_mass: f64,
    /// total_agent_payload + total_field_mass. The primary invariant to
    /// watch: with k_elim=0 this must stay exactly constant; with k_elim>0
    /// it should follow the exact closed-form elimination curve derived in
    /// the module tests, not just "decrease somehow."
    pub total_combined_mass: f64,
}

pub struct CoupledEngine {
    pub agents: Vec<Agent>,
    pub params: AgentParams,
    pub bounds: Bounds,
    pub field: ConcentrationField,
    /// Diffusion coefficient used for the field's own diffusion step. Can
    /// differ from `params.diffusion_coeff` (which governs agent motion) --
    /// physically these are different quantities (e.g. a nanoparticle's
    /// own Brownian motion vs. how fast a released small-molecule payload
    /// disperses through tissue).
    pub field_diffusion_coeff: f64,
    pub dt: f64,
}

impl CoupledEngine {
    pub fn new(
        agents: Vec<Agent>,
        params: AgentParams,
        bounds: Bounds,
        field: ConcentrationField,
        field_diffusion_coeff: f64,
        dt: f64,
    ) -> Self {
        Self { agents, params, bounds, field, field_diffusion_coeff, dt }
    }

    fn stats_for_tick(&self, tick: u64) -> CoupledTickStats {
        let total_agent_payload: f64 = self.agents.iter().map(|a| a.payload).sum();
        let total_field_mass = self.field.total_mass();
        CoupledTickStats {
            tick,
            total_agent_payload,
            total_field_mass,
            total_combined_mass: total_agent_payload + total_field_mass,
        }
    }

    /// Runs the coupled simulation for `n_ticks`. Each tick: every agent
    /// moves and releases payload (via `step_coupled`), the released
    /// amount is deposited into the field at the agent's *new* (post-
    /// motion) position, and then the field diffuses one step. Stats are
    /// recorded after each full tick (motion + release + diffusion).
    pub fn run<R: Rng + ?Sized>(&mut self, n_ticks: u64, rng: &mut R) -> Vec<CoupledTickStats> {
        let mut history = Vec::with_capacity(n_ticks as usize + 1);
        history.push(self.stats_for_tick(0));

        for tick in 1..=n_ticks {
            for agent in &mut self.agents {
                let released = agent.step_coupled(&self.params, &self.bounds, self.dt, rng);
                if released > 0.0 {
                    self.field.deposit(agent.position, released);
                }
            }
            self.field.step(self.dt, self.field_diffusion_coeff);
            history.push(self.stats_for_tick(tick));
        }

        history
    }
}

#[derive(Debug, Clone, Copy)]
pub struct FeedbackTickStats {
    pub tick: u64,
    pub total_agent_payload: f64,
    pub total_field_mass: f64,
    pub total_combined_mass: f64,
    /// Running total of mass that has left the system via true elimination
    /// (k_elim) since the start of the run. This is the exact-bookkeeping
    /// counterpart used to validate the full feedback loop: unlike the
    /// release-only case, once uptake is active there's no simple closed
    /// form for total_combined_mass(t) (uptake depends on the diffusing
    /// field, which depends on every agent's history) -- but the discrete
    /// accounting identity `total_combined_mass(t) == C0 - cumulative_eliminated(t)`
    /// must hold exactly regardless, because release/deposit and
    /// uptake/withdraw are each self-canceling transfers, and elimination
    /// is the only pathway that actually removes mass from the system.
    pub cumulative_eliminated: f64,
}

impl CoupledEngine {
    /// Runs the *full* two-way feedback loop: each tick, every agent
    /// moves, senses the local field concentration at its new position,
    /// draws an uptake amount from the field (`k_uptake * local_concentration
    /// * dt`, capped by `ConcentrationField::withdraw` at whatever mass is
    /// actually present), gains that into its payload, then undergoes
    /// exact decay split between release (deposited back into the field)
    /// and true elimination (lost). The field then takes one diffusion
    /// step.
    ///
    /// Processing is sequential per agent within a tick (not a two-phase
    /// read/write split), so two agents in the same cell in the same tick
    /// genuinely compete for the same local mass -- the second agent's
    /// `withdraw` call sees whatever the first agent's call left behind.
    /// This is a deliberate modeling choice (agents act in some sequence,
    /// not simultaneously within a tick), and it's still exactly
    /// mass-conserving regardless of the order, since every transfer is a
    /// capped, exact pairwise exchange.
    pub fn run_with_feedback<R: Rng + ?Sized>(
        &mut self,
        n_ticks: u64,
        k_uptake: f64,
        rng: &mut R,
    ) -> Vec<FeedbackTickStats> {
        let mut history = Vec::with_capacity(n_ticks as usize + 1);
        let mut cumulative_eliminated = 0.0;
        history.push(self.feedback_stats(0, cumulative_eliminated));

        for tick in 1..=n_ticks {
            for agent in &mut self.agents {
                agent.advance_position(&self.params, &self.bounds, self.dt, rng);

                let c_local = self.field.sample(agent.position);
                let desired_uptake = k_uptake * c_local * self.dt;
                let actual_uptake = self.field.withdraw(agent.position, desired_uptake);
                agent.payload += actual_uptake;

                let (released, eliminated) = agent.decay_and_release(&self.params, self.dt);
                cumulative_eliminated += eliminated;
                if released > 0.0 {
                    self.field.deposit(agent.position, released);
                }
            }
            self.field.step(self.dt, self.field_diffusion_coeff);
            history.push(self.feedback_stats(tick, cumulative_eliminated));
        }

        history
    }

    fn feedback_stats(&self, tick: u64, cumulative_eliminated: f64) -> FeedbackTickStats {
        let total_agent_payload: f64 = self.agents.iter().map(|a| a.payload).sum();
        let total_field_mass = self.field.total_mass();
        FeedbackTickStats {
            tick,
            total_agent_payload,
            total_field_mass,
            total_combined_mass: total_agent_payload + total_field_mass,
            cumulative_eliminated,
        }
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use rand::SeedableRng;
    use rand::rngs::StdRng;

    fn small_field() -> ConcentrationField {
        let bounds = Bounds::new([-10.0, -10.0, -10.0], [10.0, 10.0, 10.0]);
        ConcentrationField::new(bounds, 21, 21, 21)
    }

    /// With k_elim=0 (pure release, no true elimination), total combined
    /// mass (agents + field) must be exactly conserved across the whole
    /// run: `field.step` conserves mass exactly (proven in
    /// concentration.rs), and `step_coupled` with k_elim=0 releases
    /// exactly the agent's full loss (proven in agent.rs), so nothing
    /// should be able to leak. This is the core invariant test for the
    /// coupling itself.
    #[test]
    fn combined_mass_exactly_conserved_with_zero_elimination() {
        let bounds = Bounds::new([-10.0, -10.0, -10.0], [10.0, 10.0, 10.0]);
        let params = AgentParams { drift: [0.2, 0.0, 0.0], diffusion_coeff: 0.5, k_elim: 0.0, k_release: 0.3 };
        let agents: Vec<Agent> = (0..30).map(|_| Agent::new([0.0, 0.0, 0.0], 10.0)).collect();
        let field = small_field();
        let field_d = 0.1;
        let mut engine = CoupledEngine::new(agents, params, bounds, field, field_d, 0.01);
        let mut rng = StdRng::seed_from_u64(42);

        let history = engine.run(400, &mut rng);
        let m0 = history[0].total_combined_mass;

        for stat in &history {
            let rel_err = (stat.total_combined_mass - m0).abs() / m0;
            assert!(
                rel_err < 1e-9,
                "tick {}: combined mass {} vs initial {} (rel_err {})",
                stat.tick, stat.total_combined_mass, m0, rel_err
            );
        }
    }

    /// With both k_release>0 and k_elim>0, combined mass is NOT simply
    /// conserved (true elimination genuinely removes mass from the
    /// system) -- but it must follow the exact closed-form curve derived
    /// from the same proportional-partition argument used in agent.rs:
    ///   combined_mass(t) = C0 * (1 - (k_elim/k_total) * (1 - exp(-k_total*t)))
    /// This is exact algebra (not a fit or an approximation), so a tight
    /// tolerance is appropriate; the only slack is floating-point
    /// summation over many agents and field cells.
    #[test]
    fn combined_mass_matches_exact_elimination_formula_when_both_pathways_active() {
        let bounds = Bounds::new([-10.0, -10.0, -10.0], [10.0, 10.0, 10.0]);
        let k_elim = 0.05;
        let k_release = 0.15;
        let k_total = k_elim + k_release;
        let params = AgentParams { drift: [0.0, 0.0, 0.0], diffusion_coeff: 0.3, k_elim, k_release };

        let n_agents = 25;
        let initial_payload_each = 8.0;
        let agents: Vec<Agent> = (0..n_agents).map(|_| Agent::new([0.0, 0.0, 0.0], initial_payload_each)).collect();
        let c0 = n_agents as f64 * initial_payload_each;

        let field = small_field();
        let mut engine = CoupledEngine::new(agents, params, bounds, field, 0.1, 0.01);
        let mut rng = StdRng::seed_from_u64(7);

        let history = engine.run(600, &mut rng); // t_final = 6.0

        for stat in &history {
            let t = stat.tick as f64 * 0.01;
            let expected = c0 * (1.0 - (k_elim / k_total) * (1.0 - (-k_total * t).exp()));
            let rel_err = (stat.total_combined_mass - expected).abs() / c0;
            assert!(
                rel_err < 1e-6,
                "tick {}: combined mass {} vs exact formula {} (rel_err {})",
                stat.tick, stat.total_combined_mass, expected, rel_err
            );
        }
    }

    /// Basic wiring sanity check, independent of the exact-conservation
    /// tests above: after running with release active, the field should
    /// actually have gained mass (not stayed at zero -- which would happen
    /// if `deposit` were never actually called).
    #[test]
    fn field_actually_receives_deposited_mass() {
        let bounds = Bounds::new([-10.0, -10.0, -10.0], [10.0, 10.0, 10.0]);
        let params = AgentParams { drift: [0.0, 0.0, 0.0], diffusion_coeff: 0.2, k_elim: 0.0, k_release: 0.5 };
        let agents: Vec<Agent> = (0..10).map(|_| Agent::new([0.0, 0.0, 0.0], 5.0)).collect();
        let field = small_field();
        let mut engine = CoupledEngine::new(agents, params, bounds, field, 0.1, 0.01);
        let mut rng = StdRng::seed_from_u64(1);

        let history = engine.run(100, &mut rng);
        assert!(history[0].total_field_mass == 0.0, "field should start empty");
        assert!(
            history.last().unwrap().total_field_mass > 0.0,
            "field should have gained mass from agent release"
        );
    }

    /// No-op sanity check: with k_release=0 and k_elim=0, nothing should
    /// move mass anywhere -- combined mass stays at the initial value and
    /// field mass stays at exactly zero throughout.
    #[test]
    fn no_release_no_elimination_means_no_mass_transfer() {
        let bounds = Bounds::new([-10.0, -10.0, -10.0], [10.0, 10.0, 10.0]);
        let params = AgentParams { drift: [0.1, 0.0, 0.0], diffusion_coeff: 0.2, k_elim: 0.0, k_release: 0.0 };
        let agents: Vec<Agent> = (0..5).map(|_| Agent::new([0.0, 0.0, 0.0], 3.0)).collect();
        let field = small_field();
        let mut engine = CoupledEngine::new(agents, params, bounds, field, 0.1, 0.01);
        let mut rng = StdRng::seed_from_u64(1);

        let history = engine.run(200, &mut rng);
        for stat in &history {
            assert_eq!(stat.total_field_mass, 0.0);
            assert!((stat.total_agent_payload - 15.0).abs() < 1e-12);
        }
    }
}
